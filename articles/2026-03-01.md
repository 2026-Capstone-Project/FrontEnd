**3. 우선순위(Priority) 시스템 도입**
할 일의 중요도를 구분하기 위해 `HIGH`, `MEDIUM`, `LOW` 세 단계의 우선순위 타입을 정의하고, 이를 UI에 반영하기 위한 색상 팔레트 및 유틸리티 함수를 추가했습니다.

```typescript
// src/shared/utils/priority.ts
export function translatePriority(priority: 'HIGH' | 'MEDIUM' | 'LOW'): string {
  switch (priority) {
    case 'HIGH': return '높음'
    case 'MEDIUM': return '중간'
    case 'LOW': return '낮음'
    default: return '없음'
  }
}

// src/shared/schemas/todo.ts
const todoPriority = yup
  .mixed<PriorityType>()
  .oneOf(['HIGH', 'MEDIUM', 'LOW'], '유효하지 않은 우선순위입니다.')
```

// src/shared/hooks/common/useUnsavedCloseGuard.ts
export const useUnsavedCloseGuard = ({
  isDirty,
  onClose,
  registerCloseGuard,

  const allowCloseRef = useRef(false)
  const [isUnsavedConfirmOpen, setIsUnsavedConfirmOpen] = useState(false)
  // 변경 사항(isDirty)이 있을 경우 이탈 방지 컨펌 모달 노출
  const requestClose = useCallback(() => {
    if (isDirty && !allowCloseRef.current) {
      setIsUnsavedConfirmOpen(true)
      return
    }
    onClose()
  }, [isDirty, onClose])
