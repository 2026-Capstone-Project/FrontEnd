export const useCalendarDateRange = (view: View, date: Date) =>

        endDate: base.clone().endOf('month').format('YYYY-MM-DD'),
      }
    }
    // ... 주간/일간 뷰 대응

### [yujin5959] 인프라 설정 수정 및 인증 시스템 결함 해결

yujin5959 개발자는 개발 생산성 향상을 위한 린트 설정 수정과 사용자 경험에 직결되는 토큰 재발급 로직의 결함을 수정했습니다.

#### 1. ESLint 및 Git Hook 최적화
`lint-staged` 실행 시 캐시를 활용하도록 수정하여 커밋 속도를 개선하였으며, Husky의 `commit-msg` 실행 방식을 `pnpm exec`로 변경하여 환경 간 호환성을 확보했습니다.

```json
// package.json
"lint-staged": {
  "src/**/*.{ts,tsx}": [
    "eslint --fix --cache",
    "prettier --write"
  ]
}
```

#### 2. 401 에러 대응 및 토큰 재발급(Reissue) 로직 구현
인증 만료 시 사용자가 로그아웃되지 않고 자동으로 세션을 연장할 수 있도록 Axios 응답 인터셉터에 토큰 재발급 로직을 추가했습니다. 이는 서비스의 연속성을 보장하는 중요한 수정 사항입니다.

```typescript
// src/shared/api/axios.ts
axiosInstance.interceptors.response.use(
  (response) => response,
  async (error) => {
    const originalRequest = error.config
    if (error.response?.status === 401 && !originalRequest._retry) {
      originalRequest._retry = true
      try {
        // 쿠키 기반 토큰 재발급 요청
        await axios.post(
          `${import.meta.env.VITE_API_BASE_URL}/api/v1/security/reissue-cookie`,
          {},
          { withCredentials: true }
        )
        return axiosInstance(originalRequest)
      } catch (reissueError) {
        return Promise.reject(reissueError)
      }
    }
    return Promise.reject(error)
  }
)
```

#### 3. 라우팅 구조 고도화
설정(Settings) 관련 페이지들을 별도의 `SettingRoutes`로 분리하고 `SettingLayout`을 적용하여, 메인 서비스 로직과 설정 로직 간의 결합도를 낮추었습니다.
